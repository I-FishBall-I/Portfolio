  • //取出所有checkbox
const checkInput = Array.from(document.querySelectorAll('input'))                                    
let lastChecked = null;
    //按下shift的判斷
    function selectMany(e) {
        /*
        第一層判斷
            當第一次按下先記錄當前索引，否則清空
        第二層
            按下shift且第一次不為空則記錄第二次索引
            在對checkInput陣列進行slice(比較索引小的放前面)
            對被切出來的陣列跑foreach把每個input勾選起來
        */
        if (this.checked) {
            if (e.shiftKey && lastChecked !== null) {
                let nowCheck = checkInput.indexOf(this)
                checkInput.slice(
                    Math.min(nowCheck, lastChecked), 
                    Math.max(nowCheck, lastChecked))
                .forEach(input => input.checked = true);
            }
            lastChecked = checkInput.indexOf(this);
        } else {
            lastChecked = null;
        }
}
    //對每個checkbox加上監聽
    checkInput.forEach(input => input.addEventListener('click', selectMany));

    const
        windowTop = window.scrollY,  //畫面最頂部的位置

}

  • const item = document.querySelector('.items');
let positionX = 0;//設一個初始位置
function start(e) {
    item.classList.add('active');
    positionX = e.pageX;//將按下得位置記錄起來
}
function startMove(e) {
    e.preventDefault();
    let move = e.pageX - positionX;//移動位置=拖動到的位置-初始位置
    if (item.classList.contains('active')) {
    moved = true;
    item.scrollLeft -= move;//將item.scrollLeft減去move，就能拖曳
    positionX = e.pageX;//重新刷新初始值
    }
}
function stopMove() {
    item.classList.remove('active');
}
item.addEventListener('mousedown', start);//開始
item.addEventListener('mousemove', startMove);//移動
item.addEventListener('mouseup', stopMove);//結束
item.addEventListener('mouseleave', stopMove);//結束
//解決拖曳時如果內容有連結必須要讓他在過程中不能被觸發
document.querySelectorAll('.items a').forEach(dom => {
    dom.addEventListener('click', function (e) {
    if (moved) e.preventDefault();
    })

percent = percent * (max - min) + min; //讓速率介於0.5~2.5 =(大-小)+小    
percent > max ? max : percent < min ? min : percent;//鎖死不要讓它超過     
const text = Math.round(percent * 100) / 100;//文字
bar.textContent = `${text}×`
video.playbackRate = percent;//控制速率
}
target.addEventListener('mousemove', targetHadler)
