function selectMany(e) {
    //確認是否在兩個選中的checkbox之間
    let inBetween = false;
    /*
        如果點擊 Checkbox 2，lastChecked 被設置為 Checkbox 2。
        按住 Shift 鍵並點擊 Checkbox 4。forEach會遍歷所有的複選框
        遍歷到 Checkbox 2 時，因為它是 lastChecked，所以 inBetween 被設置為 true（表示進入範圍）。
        遍歷到 Checkbox 3 時，inBetween 已經是 true，所以 Checkbox 3 被選中。
        遍歷到 Checkbox 4 時，這個複選框是當前點擊的複選框（this），inBetween 被設置為 false（表示離開範圍）
    */
    if (e.shiftKey && this.checked) {
        checkInput.forEach((input) => {
        if (input === this || input === lastChecked) {
            inBetween = !inBetween;
        }
        //在範圍內的checkbox=true
        if (inBetween) {
            input.checked = true;
        }
        })
    }
    //更新當前的checked
    lastChecked = this;
}

Slide in on Scroll

  • 監聽scroll讓圖片滑到指定位置決定他該出現還是消失，主要用到window.scroll、offsetTop及clientHeight
  • 
window.addEventListener('scroll', checkImg)
function checkImg() {
    document.querySelectorAll('img').forEach(item => {
        const
            currentPosition = window.scrollY + innerHeight,//取得畫面最底部的位置
            imgTop = item.offsetTop + item.clientHeight / 4,//圖片1/4處
            imgBottom = item.offsetTop + item.clientHeight * 3 / 4,//圖片3/4處
            imgEnterScreen = currentPosition > imgTop,//如果底部位置超過圖片1/4處則為TRUE讓圖片進入
            imgInScreen = window.scrollY < imgBottom;//如果頂部位置超過圖片3/4處則為False讓圖片出去

        imgEnterScreen && imgInScreen ? item.classList.add('active') : item.classList.remove('active'); 
    })
}
